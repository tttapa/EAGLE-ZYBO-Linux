97  /*
98  * q = cos(ϑ / 2) + sin(ϑ / 2)·(x·i + y·j + z·k)
99  * where (x y z) is a unit vector representing the axis about which
100  * the body is rotated; ϑ is the angle by which it is rotated.
101  *
102  * (x y z) is the cross product between a vector pointing upwards (0 0 1)
103  * and the given vector; ϑ can be found using A×B = |A||B|·sin(ϑ).
104  */
105 
106  /* First check the edge case vec ~ (0 0 1). */
107  real_t eps = std::numeric_limits<real_t>::epsilon();
108  if (abs(vec[0]) <= eps && abs(vec[1]) <= eps)
109  return Quaternion::unit();
110 
111  /* Calculate the cross product and its norm. */
112  ColVector<3> cross = {vec[1], -vec[0], 0};
113  real_t crossNorm = norm(cross);
114  cross /= crossNorm;
115 
116  /* Calculate the angle ϑ. */
117  real_t angle = std::asin(crossNorm / norm(vec));
118 
119  /* Calculate the resulting quaternion. */
120  return vcat(std::cos(angle / 2), std::sin(angle / 2) * cross);
