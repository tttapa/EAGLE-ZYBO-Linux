83 int rmem(uint64_t offset) {
84  int i, fd;
85  int value;
86  uint64_t base;
87  int dump = 1;
88  int cached = 0;
89  volatile uint8_t *mm;
90  fd = open("/dev/mem", O_RDWR | (!cached ? O_SYNC : 0));
91  if (fd < 0) {
92  fprintf(stderr, "open(/dev/mem) failed (%d)\n", errno);
93  return 1;
94  }
95  //offset = OFFSET_THRUST;
96  //value = thrust;
97  base = offset & PAGE_MASK;
98  offset &= ~PAGE_MASK;
99 
100  mm = mmap(NULL, PAGE_SIZE, PROT_READ | PROT_WRITE, MAP_SHARED, fd, base);
101  if (mm == MAP_FAILED) {
102  fprintf(stderr, "mmap64(0x%x@0x%lx) failed (%d)\n", PAGE_SIZE, base,
103  errno);
104  }
105  for (i = 0; i < dump; i++) {
106  value = *(volatile uint32_t *) (mm + offset + i * 4);
107  //printf("0x%016llx = 0x%08lx\n", (base + offset + i*4), value);
108  }
109 
110  munmap((void *) mm, PAGE_SIZE);
111  close(fd);
112  return value;
113 }
