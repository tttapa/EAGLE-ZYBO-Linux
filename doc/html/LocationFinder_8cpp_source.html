70  // ----------------------------
71  try {
72  angle_t newangle = sq.getAngle();
73  angle = angleTracker.update(newangle);
74  } catch (std::runtime_error &) {
75  // Fails if we can't determine the angle (when there are no lines)
76  }
77 
78  // Find the center of the square and the average length of the sides
79  // -----------------------------------------------------------------
80  Vec2f center;
81  auto points = sq.points;
82 
83  // === No points ===
84  // If we found just one line, no points → nothing can be done
85  if (!points[0] && !points[1])
86  return Point::invalid();
87  angle_t diagAngle = angle + 45_deg;
88  // === One point ===
89  // If only the first point is valid
90  if (points[0] && !points[1] && !points[2] && !points[3]) {
91  Vec2f diag = {diagAngle.cosf(), diagAngle.sinf()};
92  center = sideLen / sqrt(2) * diag + points[0]->vec();
93  }
94  // If only the second point is valid
95  else if (!points[0] && points[1] && !points[2] && !points[3]) {
96  Vec2f diag = {diagAngle.cosf(), diagAngle.sinf()};
97  center = sideLen / sqrt(2) * diag + points[1]->vec();
98  }
99  // === Three points ===
100  // If only the first two points are valid
101  else if (points[0] && points[1] && !points[2] && !points[3]) {
102  sideLen = Point::distance(*points[0], *points[1]);
103  Vec2f diag = {diagAngle.cosf(), diagAngle.sinf()};
104  center = sideLen / sqrt(2) * diag + points[0]->vec();
105  }
106  // === Four points ===
107  // If we found all four points
108  else if (points[0] && points[1] && points[2] && points[3]) {
109  sideLen = (Point::distance(*points[0], *points[1]) +
110  Point::distance(*points[0], *points[2]) +
111  Point::distance(*points[1], *points[3]) +
112  Point::distance(*points[2], *points[3])) /
113  4;
114  center = Point::average(Point::average(*points[0], *points[1]),
115  Point::average(*points[2], *points[3]));
116  } else {
117  std::cerr << ANSIColors::redb
118  << "Error: Somehow, we only found three points"
119  << ANSIColors::reset << std::endl;
120  return Point::invalid();
121  }
