36  /// When this many consecutive black pixels are encountered, stop following
37  /// the line.
38  constexpr static uint HOUGH_MAX_GAP = 64;

40  /**
41  * @brief Starting from the given pixel, move in the direction of the
42  * given angle, and count the number of white pixels, weighing them
43  * more as the distance increases. When more than #HOUGH_MAX_GAP
44  * consecutive black pixels are encountered, stop looking any
45  * farther.
46  * @param px
47  * The starting point.
48  * @param angle
49  * The direction to look in.
50  */
51  HoughResult hough(Pixel px, angle_t angle) const {
52  BresenhamLine line = {px, angle, width(), height()};
53  uint previousWhite = 0;
54  while (line.hasNext()) {
55  Pixel point = line.next();
56  // If pixel is white
57  if (get(point))
58  previousWhite = line.getCurrentLength();
59  else if (line.getCurrentLength() - previousWhite >= HOUGH_MAX_GAP)
60  break;
61  }
62  return {angle, previousWhite};
63  }

65  /**
66  * @brief Scan 360° around a given pixel, and find the angle with the most
67  * white pixels. This version is not very accurate, as it doesn't
68  * handle the case where multiple angles have the same maximum
69  * count.
70  * On the other hand, it's a bit faster than the more accurate
71  * alternatives, making it a good candidate for finding the first
72  * angle estimate, to narrow down the search space.
73  */
74  HoughResult findLineAngle(Pixel px) const {
75  std::array<HoughResult, angle_t::resolution()> houghRes;
76  for (uint i = 0; i < angle_t::resolution(); ++i)
77  houghRes.at(i) = hough(px, angle_t(i));
78  return *std::max_element(houghRes.begin(), houghRes.end());
79  }

116  /**
117  * @brief Similar to #findLineAngle, but is more accurate, because
118  * it uses the average of the range of angles with the hightest
119  * vote counts. It also doesn't scan 360°, but takes 2@p N samples
120  * around a given center angle.
121  */
122  template <uint N>
123  HoughResult findLineAngleAccurateRange(Pixel px,
124  angle_t centerAngle) const {
125  static_assert(2 * N < angle_t::resolution());
126  std::array<HoughResult, 2 * N + 1> houghRes;
127  uint centerAngleIndex = centerAngle.getIndex();
128 
129  // If the range contains no discontinuities
130  // ├───┬───┬───╤═══╤═══╤═══╪═══╤═══╤═══╤───┬───┬───┤
131  // 0 C-N C C+N R-1
132  // C = center
133  // N = number of steps around the center
134  // R = resolution (R steps = 720°)
135  if (centerAngleIndex - N < angle_t::resolution() &&
136  centerAngleIndex + N < angle_t::resolution()) {
137  uint initialAngleIndex = centerAngleIndex - N;
138  for (uint i = 0; i <= 2 * N; ++i)
139  houghRes.at(i) = hough(px, angle_t(initialAngleIndex + i));
140  }
141  // If the 0-2π discontinuity is in the first part of the range
142  // (< centerAngle)
143  // ╞═══╪═══╤═══╤═══╤───┬───┬───┬───┬───┬───┬───╤═══╡
144  // 0 C C+N C-N R-1
145  else if (centerAngleIndex + N < angle_t::resolution()) {
146  uint initialAngleIndex =
147  angle_t::resolution() + centerAngleIndex - N;
148  uint i = 0;
149  for (uint j = initialAngleIndex; j < angle_t::resolution();
150  ++i, ++j)
151  houghRes.at(i) = hough(px, angle_t(j));
152  for (uint j = 0; j <= centerAngleIndex + N; ++i, ++j)
153  houghRes.at(i) = hough(px, angle_t(j));
154  }
155  // If the 0-2π discontinuity is in the second part of the range
156  // (> centerAngle)
157  // ╞═══╤───┬───┬───┬───┬───┬───┬───╤═══╤═══╤═══╪═══╡
158  // 0 C+N C-N C R-1
159  else if (angle_t::resolution() - N < angle_t::resolution()) {
160  uint initialAngleIndex = centerAngleIndex - N;
161  uint i = 0;
162  for (uint j = initialAngleIndex; j < angle_t::resolution();
163  ++i, ++j)
164  houghRes.at(i) = hough(px, angle_t(j));
165  for (uint j = 0; j <= N + centerAngleIndex - angle_t::resolution();
166  ++i, ++j)
167  houghRes.at(i) = hough(px, angle_t(j));
168  } else {
169  throw std::runtime_error("Something is very wrong");
170  }
171  // If there are multiple elements with the same maximum count,
172  // max_element will return an iterator to the first element with this
173  // count
174  auto max = std::max_element(houghRes.begin(), houghRes.end());
175 
176  // Find the first element after the first maximum that has a lower count
177  // than ⅞ of the maximum count
178  // and then subtract one to get the last element that has the maximum
179  // count.
180  // If no element with a lower count is found, find_if returns
181  // houghRes.end(), which means that last_max will be an iterator to
182  // the last element of the array.
183  // In that case, there's probably something wrong: the line might be
184  // too thick, and all counts within the specified range are very high.
185 
186  // predicate res.count < 7 * max->count / 8;
187  auto predicateLessThanSevenEighths = [&](auto res) {
188  return res.count < max->count - max->count / 8;
189  };
190  // std::find_if returns an iterator the first element that satisfies the
191  // predicate
192  //
193  auto first_max =
194  std::find_if(std::make_reverse_iterator(max), //
195  houghRes.rend(), predicateLessThanSevenEighths);
196  auto last_max =
197  std::find_if(max, //
198  houghRes.end(), predicateLessThanSevenEighths);
199 
200  if (first_max == houghRes.rend() || last_max == houghRes.end())
201  cerr << ANSIColors::yellow
202  << "Warning: angle with maximum count could lie outside of "
203  "the specified range\r\n"
204  << ANSIColors::reset;
205 
206  if (first_max != houghRes.rbegin())
207  --first_max;
208  if (last_max != houghRes.begin())
209  --last_max;
210 
211  return {
212  angle_t::average(first_max->angle, last_max->angle),
213  max->count,
214  };
215  }

234  /**
235  * @brief Get the width of the line at a given point.
236  *
237  * Start at the given pixel, and move in the given direction for
238  * @p maxLineGap pixels, in each pixel, move in the perpendicular direction
239  * until a black pixel is encountered. At this point, the half width of the
240  * line from the base line is known.
241  * Finally, the maximum of all distances is returned.
242  *
243  * If the width at any point exceeds the #MAX_LINE_WIDTH, #MAX_LINE_WIDTH is
244  * returned.
245  *
246  * @param pixel
247  * A point on the line.
248  * @param angle
249  * The slope of the line.
250  * @param maxLineGap
251  * The maximum gap that could be in the line. Do this many width
252  * samples, and return the maximum.
253  * @param plus90deg
254  * Should 90° be added or subtracted to get the perpendicular
255  * direction?
256  * @return
257  * The maximum half width of the line, over the @p maxLineGap
258  * pixels along the given line.
259  */
260  uint getWidthAtPointOnLine(Pixel pixel, CosSin angle,
261  uint maxLineGap = MAX_GAP,
262  bool plus90deg = true) const {
263  BresenhamLine alongLine = {pixel, angle, width(), height()};
264  CosSin perpendicularAngle = angle.perpendicular(plus90deg);
265  uint maxWidthSoFar = 0;
266  // Follow a path along the given line for maxLineGap pixels
267  for (uint i = 0; i <= maxLineGap && alongLine.hasNext(); ++i) {
268  Pixel pixelAlongLine = alongLine.next();
269  // For each pixel along this path, move away from the line,
270  // perpendicular to it, until you find a black pixel, until you
271  // fall off the canvas, or until the maximum line width is exceeded.
272  BresenhamLine perpendicular = {pixelAlongLine, perpendicularAngle,
273  width(), height()};
274  while (perpendicular.hasNext() &&
275  perpendicular.getCurrentLength() <= MAX_LINE_WIDTH) {
276  Pixel pixel = perpendicular.next();
277  if (get(pixel) == 0x00)
278  break;
279  }
280  if (perpendicular.getCurrentLength() > MAX_LINE_WIDTH)
281  // throw std::runtime_error("Error: Endless loop detected");
282  return MAX_LINE_WIDTH;
283 
284  // If we found a black pixel before exceeding the maximum line width
285  if (perpendicular.getCurrentLength() > maxWidthSoFar)
286  // getCurrentLength() gives the length to the first black pixel,
287  // so that's one pixel too much. Fix at the end of the function.
288  maxWidthSoFar = perpendicular.getCurrentLength();
289  }
290  return maxWidthSoFar - 1;
291  }

293  /**
294  * @brief Find the middle of a line.
295  *
296  * @param pointOnLine
297  * A point that lies on the line.
298  * @param lineAngle
299  * An estimate of the slope of the line.
300  * @param maxLineGap
301  * The maximum gap of black pixels that can be recovered.
302  * @return // TODO
303  */
304  optional<GetMiddleResult> getMiddle(Pixel pointOnLine, CosSin lineAngle,
305  uint maxLineGap = MAX_GAP) const {
306  // If the given point does not lie on a line, we can't find the middle
307  // of the line
308  if (get(pointOnLine) == 0x00)
309  return std::nullopt;
310 
311  CosSin oppositeAngle = lineAngle.opposite();
312  // Get the half width of the line in four directions.
313  // U2 U1
314  // ↑↑↑ ↑↑↑
315  // ---o---> line
316  // ↓↓↓ ↓↓↓
317  // L2 L1
318  uint widthUpper1 = getWidthAtPointOnLine(pointOnLine, lineAngle, //
319  maxLineGap / 2, true);
320  uint widthLower1 = getWidthAtPointOnLine(pointOnLine, lineAngle, //
321  maxLineGap / 2, false);
322  uint widthUpper2 = getWidthAtPointOnLine(pointOnLine, oppositeAngle, //
323  maxLineGap / 2, false);
324  uint widthLower2 = getWidthAtPointOnLine(pointOnLine, oppositeAngle, //
325  maxLineGap / 2, true);
326 
327  // Use the maximum of the upper width and the maximum of the lower width
328  uint widthUpper = std::max(widthUpper1, widthUpper2);
329  uint widthLower = std::max(widthLower1, widthLower2);
330 
331  // If either of the widths exceeds the #MAX_LINE_WIDTH, the result is
332  // invalid: it is probably a large white spot, not a thin line
333  if (widthUpper >= MAX_LINE_WIDTH || widthLower >= MAX_LINE_WIDTH)
334  return std::nullopt;
335 
336  // Use the difference between the two widths to move the given point
337  // towards the center
338  int middlePointCorrection_x2 = widthUpper - widthLower;
339  uint middlePointCorrDistance = std::abs(middlePointCorrection_x2) / 2;
340  // Should we move up or down? (+90° or -90° from the estimated angle)
341  bool corrDirection = middlePointCorrection_x2 > 0;
342  CosSin corrAngle = lineAngle.perpendicular(corrDirection);
343  Pixel middle = move(pointOnLine, corrAngle, middlePointCorrDistance);
344  return GetMiddleResult{middle, widthUpper + widthLower - 1};
345  }

521  /// Don't use vertical lines as the first line, as this can result in
522  /// finding a first square that's not in the center of the frame.
523  constexpr static uint MAXIMUM_VERTICAL_START_LINE_WIDTH = 32; // TODO
524 
525  /**
526  * @brief Find the vertical range of white pixels closest to the center of
527  * the given column, then try to find an estimate of the line going
528  * through the center of this range.
529  *
530  * ### Why near-vertical lines are a problem
531  *
532  * Consider the following example: all letters are white pixels, all dots
533  * are black. The arrow indicates the column we're searching in, and `C` is
534  * the center of the frame.
535  * ```
536  * ↓
537  * X X X X X X X
538  * · · X · · · ·
539  * · · X · · · ·
540  * · · X · · · ·
541  * X X X X X X X
542  * · · X · · · ·
543  * · · · C · · ·
544  * · · · R · · ·
545  * X X X R X X X
546  * · · · M · · ·
547  * · · · R · · ·
548  * · · · R · · ·
549  * X X X X X X X
550  * ```
551  * We're looking for the range of white pixels in the indicated column,
552  * closest to the center `C`. This turns out to be the range labeled with
553  * `R` (including `C`). The middlepoint of that range is `M`, so that'll be
554  * the starting point for our first line.
555  * However, notice how we ended up in a square that's farther away from the
556  * center than we'd like. The square above it is safer, because the bottom
557  * or top edge of squares far away from the center might not be in the
558  * frame.
559  *
560  * ### Solution
561  *
562  * To get around this problem, a maximum vertical range length is defined,
563  * so the algorithm ignores near-vertical lines, and finds only horizontal
564  * and diagonal lines. If it does find a vertical line, it just moves
565  * sideways, to a different column, where it will find a horizontal line.
566  *
567  * @param x
568  * The x-coordinate of the column to search in.
569  * @return
570  * Returns no result if the vertical range of white pixels is too
571  * long, which would indicate a vertical line.
572  * Also returns no result if #getFirstLineEstimate fails, or if no
573  * white pixels are found in the given column.
574  */
575  optional<FirstLineEstimate> getFirstLineEstimate(uint x) const {
576  if (x >= width())
577  throw std::out_of_range("x out of range");
578 
579  // Iterate like this: `... 6 4 2 0 1 3 5 ...` to find the pixels
580  // close to the center first.
581  CenterPointOutLineIterator c = height();
582 
583  // If the center pixel is white, look for the first black pixels in both
584  // directions
585  if (get(x, c.getCenter())) {
586  uint y = c.getCenter();
587  uint first_white = y;
588  // Find the lowest white pixel of the line through the center
589  while (y < height()) {
590  if (get(x, y))
591  first_white = y;
592  else
593  break;
594  --y;
595  }
596  // Find the highest white pixel of the line through the center
597  uint last_white = first_white;
598  y = c.getCenter();
599  while (y < height()) {
600  if (get(x, y))
601  last_white = y;
602  else
603  break;
604  ++y;
605  }
606  y = (first_white + last_white) / 2;
607  assert(last_white >= first_white);
608  if (last_white - first_white >= MAXIMUM_VERTICAL_START_LINE_WIDTH)
609  return std::nullopt;
610  else
611  return getFirstLineEstimate({x, y});
612  }
613  // If the center pixel is not white, look for the first white pixel in
614  // both directions
615  uint first_white = height();
616  while (c.hasNext()) {
617  uint y = c.next();
618  if (get(x, y)) {
619  first_white = y;
620  break;
621  }
622  }
623  if (first_white >= height())
624  return std::nullopt;
625 
626  uint last_white = first_white;
627  // If the first white pixel is below the center
628  if (first_white < c.getCenter()) {
629  // Look downwards for the last white pixel of that line
630  uint y = first_white;
631  while (y < height()) {
632  if (get(x, y))
633  last_white = y;
634  else
635  break;
636  --y;
637  }

724  /**
725  * @brief Follow the given line to find the line perpendicular to it.
726  * Move along the line for @p minDistance pixels, then move away
727  * from the line, perpendicular to it for twice its width plus the
728  * given offset. Then move from this point parallel to the original
729  * line, and search for the next white pixel, which is probably
730  * part of the line perpendicular to the given line.
731  *
732  * ```
733  * minDistance
734  * / \
735  * +-----s---> search path
736  * | 2×width + offset
737  * ---o---------> line
738  * ```
739  *
740  * @param line
741  * The line to follow.
742  * @param direction
743  * Should 90° be added or subtracted to get the perpendicular
744  * direction?
745  * @param offset
746  * The offset of the search path, measured perpendicularly from the
747  * @param minDistance
748  * The minimum distance between the given center point of the line
749  * and the perpendicular line. Essentially moves the starting point
750  * of the search along the line for this distance.
751  * line.
752  * @return // TODO
753  */
754  optional<LineResult> findNextLine(LineResult line, bool direction,
755  uint offset = 0,
756  uint minDistance = 0) const {
757  angle_t angle = line.angle;
758  angle_t perp = angle.perpendicular(direction);
759 
760  Pixel searchStart = line.lineCenter;
761  searchStart = move(searchStart, perp, 2 * line.width + offset);
762  if (minDistance)
763  searchStart = move(searchStart, line.angle, minDistance);
764 
765  uint minWidth = line.width / 3;
766 
767  BresenhamLine path = {searchStart, angle, width(), height()};
768  Pixel pixel;
769  do {
770  // Follow the search path
771  pixel = path.next();
772  // Find the first white pixel
773  while (path.hasNext() && get(pixel) == 0x00)
774  pixel = path.next();
775  Pixel firstWhite = pixel;
776  // Find the first black pixel
777  while (path.hasNext() && get(pixel) != 0x00)
778  pixel = path.next();
779  Pixel firstBlack = pixel; // TODO: OBOE, but I don't really care
780  // In the center of this range of white pixels, try to find the
781  // perpendicular line
782  Pixel middle = Pixel::average(firstBlack, firstWhite);
783  optional<LineResult> possibleLine =
784  checkLine(middle, perp, minWidth);
785  // If a line has been found, return it
786  if (possibleLine)
787  return possibleLine;
788  // Otherwise, keep looking along the search path
789  } while (path.hasNext());
790  // If we reach the end of the frame without finding a perpendicular line
791  // return no result, so the calling function can call again with a
792  // different offset (or give up)
793  return std::nullopt;
794  }

839  /**
840  * @brief This is the main function: it looks for a complete square of the
841  * grid, as close to the center of the frame as possible.
842  *
843  * @image html square.png
844  * @param initialTries (default: 1)
845  * When finding the first two corners, this many Bresenham lines will
846  * run parallel to the initial line in order to prevent a bad result
847  * because of a small gap in the line.
848  * @param initialTriesFactor (default: 2.0)
849  * When finding the first two corners, the parallel Bresenham lines
850  * be separated by this factor multiplied by the initial line width.
851  *
852  * @return // TODO
853  */
854  Square findSquare(uint initialTries = 1, float initialTriesFactor = 2.0f) {
855  assert(initialTries >= 1);
856  assert(initialTries > 0.0f);
857  Square sq = {};
858 
859  try {
860  // Get the line closest to the center of the frame.
861  // the result is an array of two half-lines, approximately opposite
862  // of each other
863  auto firstLines = getFirstTwoHalfLines();
864  sq.lines[0] = firstLines[0];
865  sq.lines[1] = firstLines[1];
866 
867  // Return empty square if we couldn't find the first line.
868  if (!sq.lines[0].has_value())
869  return sq;
870 
871  // Determine the direction to turn in (+90° or -90°)
872  // Check if the first line lies left or right of the center of the frame
873  // and pick the direction that will result in the square containing the
874  // center point (if possible)
875  bool direction = false;
876  Line mathLine = {sq.lines[0]->lineCenter, sq.lines[0]->angle};
877  direction = mathLine.leftOfPoint(center());
878 
879  // Try finding the first two corners multiple times. We'll remember the
880  // ones closest to the initial point, so if there's a hole in the image's
881  // line, one of the Bresenham lines will still likely find the correct
882  // perpendicular line.
883  bool firstCornerFound = false;
884  bool secondCornerFound = false;
885  float dist1 = 0.0, dist2 = 0.0, currentDistance;
886  uint jump1 = 0, jump2 = 0;
887  Point initialPoint = sq.lines[0]->lineCenter;
888  Point tempPoint;
889  std::optional<LineResult> tempLine;
890  for (uint i = 0; i < initialTries; i++) {
891 
892  // Jump initialTriesFactor times the initial line width.
893  uint jump =
894  i * std::round(initialTriesFactor * sq.lines[0]->width);
895 
896  // First corner: remember closest point to the initial point.
897  tempLine = findNextLine(sq.lines[0], direction, jump);
898  if (tempLine.has_value()) {

900  currentDistance =
901  Point::distanceSquared(initialPoint, tempPoint);
902  if (!firstCornerFound || currentDistance < dist1) {
903  firstCornerFound = true;
904  dist1 = currentDistance;
905  sq.points[0] = tempPoint;
906  sq.lines[2] = tempLine;
907  jump1 = jump;
908  }
909  }
910 
911  // Second corner: remember closest point to the initial point.
912  tempLine = findNextLine(sq.lines[1], !direction, jump);
913  if (tempLine.has_value()) {
914  tempPoint = intersect(*sq.lines[1], *tempLine);
915  currentDistance =
916  Point::distanceSquared(initialPoint, tempPoint);
917  if (!secondCornerFound || currentDistance < dist2) {
918  secondCornerFound = true;
919  dist2 = currentDistance;
920  sq.points[1] = tempPoint;
921  sq.lines[3] = tempLine;
922  jump2 = jump;
923  }
924  }
925  }
926 
927  // If we found all lines so far (first two, second, and third)
928  if (firstCornerFound && secondCornerFound) {
929 
930  // Calculate the distance between the first two points
931  // (note that this is not the Euclidian distance, but the
932  // number of pixels that would be drawn by Bresenham)
933  uint minDistance = std::floor(
934  std::max(std::abs(sq.points[0]->x - sq.points[1]->x),
935  std::abs(sq.points[0]->y - sq.points[1]->y)));
936  // Use 3/4 of this distance as a minimum distance for where to
937  // start looking for the fourth line, as it's very likely that
938  // the other sides of the square are roughly the same length as
939  // the first side
940  minDistance -= minDistance / 4;

942  const uint maxOffset = minDistance / 2;
943  const uint offsetIncr =
944  std::max(sq.lines[2]->width, sq.lines[3]->width);
945

947  while (!sq.lines[4].has_value() && offset < maxOffset) {
948  sq.lines[4] = // find the fourth line along the second
949  findNextLine(sq.lines[2], direction, offset,
950  std::max(minDistance - jump1, 0u));
951  if (!sq.lines[4].has_value()) // if not found along second
952  sq.lines[4] = // find the fourth line along the third
953  findNextLine(sq.lines[3], !direction, offset,
954  std::max(minDistance - jump2, 0u));
955  // next time, try again with a different offset
956  offset += offsetIncr;
957  }
958  // If we found a fourth line, calculate the intersections
959  if (sq.lines[4].has_value()) {
960  sq.points[2] = intersect(*sq.lines[2], *sq.lines[4]);
961  sq.points[3] = intersect(*sq.lines[3], *sq.lines[4]);
962  }
963  } else if (sq.lines[2].has_value()) {

965  } else if (sq.lines[3].has_value()) {
966  // TODO
967  } else {
968  // TODO
969  }
970  } catch (std::exception &e) {
971  cerr << ANSIColors::redb << e.what() << ANSIColors::reset << endl;
972  }
