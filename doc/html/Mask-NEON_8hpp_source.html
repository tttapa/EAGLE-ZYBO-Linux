9  * @brief Checks whether the given RGB pixels are within a certain range in
10  * HSV space.
11  *
12  * @f$
13  * \begin{align}
14  * M &= \max(R, G, B) \\
15  * m &= \min(R, G, B) \\
16  * \Delta &= M - m \\
17  * \end{align}
18  * @f$
19  *
20  * **Hue**
21  * @f$
22  * H =
23  * \begin{cases}
24  * 60° (G - B) / \Delta & M = R \\
25  * 60° (B - R) / \Delta + 120° & M = G \\
26  * 60° (R - G) / \Delta + 240° & M = B
27  * \end{cases} \\
28  * \begin{align}
29  * \left|H\right| \quad<&\quad H_{thres} \\
30  * \Leftrightarrow&\quad M = R \quad\wedge\quad
31  * 60° \frac{\left|G - B\right|}{\Delta} \;<\; H_{thres} \\
32  * \Leftrightarrow&\quad M = R \quad\wedge\quad
33  * \left|G - B\right| \;<\; \frac{\Delta}{60° / H_{thres}} \\
34  * \Leftrightarrow&\quad M = R \quad\wedge\quad
35  * \left|G - B\right|
36  * \;<\; \Delta \gg \log_2\left(\frac{60°}{H_{thres}}\right) \\
37  * \end{align}
38  * @f$
39  *
40  * **Saturation**
41  * @f$
42  * \begin{align}
43  * S \quad=&\quad 255 \frac{\Delta}{M} \;>\; S_{thres} \\
44  * \Leftrightarrow&\quad \Delta \;>\; \frac{M}{255/S_{thres}} \\
45  * \Leftrightarrow&\quad \Delta \;>\; M \gg \left(8 - \log_2\left(S_{thres}\right)\right)
46  * \end{align}
47  * @f$
48  *
49  * **Value**
50  * @f$
51  * V \quad=\quad M \;>\; V_{thres}
52  * @f$
53  *
54  * @image html Colors-Mask-Light.png
55  * @image html Colors-Mask-Dark.png
56  * @image html Colors-Mask-Gradient.png
57  *
58  * @todo Update this documentation for the fourth condition.
59  *
60  * @param colors
61  * A pointer to an array of at least 16 RGB pixels (or 48 bytes).
62  */
63 inline uint8x16_t applyMask(const uint8_t *colors) {
64  // First, load 48 bytes of strided RGB data from the image in three vector
65  // registers, one for red, one for green, and one for blue.
66  uint8x16x3_t rgb = vld3q_u8(colors);
67  uint8x16_t r = rgb.val[0];
68  uint8x16_t g = rgb.val[1];
69  uint8x16_t b = rgb.val[2];
70 
71  // Calculate the maximum and minimum of the RGB values, and their difference
72  uint8x16_t max = vmaxq_u8(vmaxq_u8(r, g), b);
73  uint8x16_t min = vminq_u8(vminq_u8(r, g), b);
74  uint8x16_t delta = vsubq_u8(max, min);
75 
76  // Condition for hue: M == R && Δ / (60 / thres) > |G - B|
77  // bool hue_cond = (r == max) && (delta >> HUE_THRES_SHIFT > abs(g - b));
78  uint8x16_t hue_cond =
79  vandq_u8(vceqq_u8(r, max),
80  vcgtq_u8(vshrq_n_u8(delta, HUE_THRES_SHIFT), vabdq_u8(g, b)));
81 
82  // Condition for saturation: Δ > M / (256 / thres)
83  // bool sat_cond = delta > (max >> SATURATION_THRES_SHIFT);
84  uint8x16_t sat_cond =
85  vcgtq_u8(delta, vshrq_n_u8(max, SATURATION_THRES_SHIFT));
86 
87  // Condition for value/brightness: V > thres
88  // bool val_cond = max > BRIGHTNESS_THRES;
89  uint8x16_t val_cond = vcgtq_u8(max, vdupq_n_u8(BRIGHTNESS_THRES));
90 
91  // TODO: comment
92  uint8x16_t valsat_cond = vcgtq_u8(
93  vhaddq_u8(vmulq_u8(vshrq_n_u8(max, 4), vshrq_n_u8(max, 5)), delta),
94  vshrq_n_u8(max, 2));
95 
96  // All conditions must be satisfied, so AND them all together.
97  return vandq_u8(vandq_u8(sat_cond, val_cond),
98  vandq_u8(hue_cond, valsat_cond));
99 }
