21 inline int8x16_t applyMaskFunction(const uint8_t *colors) {
22  // First, load 48 bytes of strided BGR data from the image in three vector
23  // registers, one for red, one for green, and one for blue.
24  uint8x16x3_t bgr = vld3q_u8(colors);
25  // Shift the red vector 2 bits to the right, to divide by 4.
